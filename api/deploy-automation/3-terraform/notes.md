1. Create main.tf and paste this into content:
```commandline
terraform {
  required_providers {
    aws = {
      source = "hashicorp/aws"
    }
  }
}

resource "aws_instance" "ec2_with_terraform" {
  ami           = var.ami
  instance_type = var.instance_type

  network_interface {
    network_interface_id = var.network_interface_id
    device_index         = 0
  }

  credit_specification {
    cpu_credits = "unlimited"
  }
}
```
2. Add this into variables.tf:
```commandline
variable "network_interface_id" {
  type = string
  default = "eni-0efe8bdb1520ef82c"
}

variable "ami" {
    type = string
    default = "ami-0f5e8a042c8bfcd5e"
}

variable "instance_type" {
    type = string
    default = "t2.micro"
}
```
3. Run
```commandline
terraform init
terraform plan
terraform apply --auto-approve
```
4. To clean up, run `terraform destroy`  

Ref: https://www.techtarget.com/searchcloudcomputing/tip/How-to-launch-an-EC2-instance-using-Terraform
